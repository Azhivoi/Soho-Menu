function buildAddress() {
const street = document.getElementById('clientStreet').value;
const house = document.getElementById('clientHouse').value;
const entrance = document.getElementById('clientEntrance').value;
const floor = document.getElementById('clientFloor').value;
const apartment = document.getElementById('clientApartment').value;
let address = street;
if (house) address += ', д. ' + house;
if (entrance) address += ', под. ' + entrance;
if (floor) address += ', этаж ' + floor;
if (apartment) address += ', кв. ' + apartment;
return address;
}
function parseAddress(address) {
if (!address) return {};
const parts = {};
// Try to extract house
const houseMatch = address.match(/,\s*д\.\s*(\S+)/);
if (houseMatch) parts.house = houseMatch[1];
// Try to extract entrance
const entranceMatch = address.match(/,\s*под\.\s*(\S+)/);
if (entranceMatch) parts.entrance = entranceMatch[1];
// Try to extract floor
const floorMatch = address.match(/,\s*этаж\s*(\S+)/);
if (floorMatch) parts.floor = floorMatch[1];
// Try to extract apartment
const aptMatch = address.match(/,\s*кв\.\s*(\S+)/);
if (aptMatch) parts.apartment = aptMatch[1];
// Street is everything before first comma
const streetMatch = address.match(/^([^,]+)/);
if (streetMatch) parts.street = streetMatch[1].trim();
return parts;
}

price: Number(item.price),
qty: item.quantity,

discountBtn.classList.add('active');
} else {
discountBtn.textContent = '%';
discountBtn.classList.remove('active');
}

updateTotal();

}

function formatPhoneNumber(phone) {
let digits = phone.replace(/\D/g, '');
if (digits.startsWith('375')) digits = digits.substring(3);
if (digits.startsWith('80')) digits = digits.substring(2);
// Return just the number part without +375 prefix
if (digits.length >= 9) {
return `(${digits.substring(0, 2)}) ${digits.substring(2, 5)}-${digits.substring(5, 7)}-${digits.substring(7, 9)}`;
}
return digits;
}

// Phone mask setup
function setupPhoneMask() {
const phoneInput = document.getElementById('clientPhone');
if (!phoneInput) return;
const PREFIX = '+375 ';
phoneInput.addEventListener('focus', function() {
if (!this.value || this.value.length < PREFIX.length) {
this.value = PREFIX;
}
});
phoneInput.addEventListener('blur', function() {
if (this.value === PREFIX) {
this.value = '';
}
});
phoneInput.addEventListener('input', function(e) {
// Always ensure prefix is there
if (!this.value.startsWith(PREFIX)) {
this.value = PREFIX;
return;
}

// Get only the digits after prefix
let value = this.value.substring(PREFIX.length).replace(/\D/g, '');

// Search customer when we have enough digits
if (value.length >= 9) {
searchCustomerByPhone(this.value);
}
// Limit to 9 digits
value = value.substring(0, 9);
// Format as user types: (XX) XXX-XX-XX
let formatted = PREFIX;
if (value.length > 0) {
formatted += '(' + value.substring(0, Math.min(2, value.length));
if (value.length >= 2) {
formatted += ')';
if (value.length > 2) {
formatted += ' ' + value.substring(2, Math.min(5, value.length));
if (value.length >= 5) {
formatted += '-' + value.substring(5, Math.min(7, value.length));
if (value.length >= 7) {
formatted += '-' + value.substring(7, 9);
}
}
}
}
}
this.value = formatted;
});
phoneInput.addEventListener('keydown', function(e) {
const cursorPos = this.selectionStart;
// Don't allow deleting the prefix
if (cursorPos <= PREFIX.length && (e.keyCode === 8 || e.keyCode === 46)) {
if (this.selectionEnd - this.selectionStart === 0) {
e.preventDefault();
return;
}
}
// Allow: backspace, delete, tab, escape, enter
if ([46, 8, 9, 27, 13].indexOf(e.keyCode) !== -1 ||
// Allow: Ctrl+A, Ctrl+C, Ctrl+V, Ctrl+X
(e.keyCode === 65 && e.ctrlKey === true) ||
(e.keyCode === 67 && e.ctrlKey === true) ||
(e.keyCode === 86 && e.ctrlKey === true) ||
(e.keyCode === 88 && e.ctrlKey === true) ||
// Allow: home, end, left, right
(e.keyCode >= 35 && e.keyCode <= 39)) {
return;
}
// Ensure that it is a number and stop the keypress
if ((e.shiftKey || (e.keyCode < 48 || e.keyCode > 57)) && (e.keyCode < 96 || e.keyCode > 105)) {
e.preventDefault();
}
});
}

// Cache for loaded category products
const categoryProductsCache = {};
async function loadMenu() {
try {
// Try cache first (5 min TTL)
const cached = localStorage.getItem('soho_menu_cache');
const cacheTime = localStorage.getItem('soho_menu_cache_time');
if (cached && cacheTime && (Date.now() - parseInt(cacheTime)) < 300000) {
const data = JSON.parse(cached);
categories = data.categories;
window.allMenuProducts = data.products || [];
renderCategories();
if (categories.length > 0) await loadCategoryProducts(categories[0].id);
return;
}
// Load categories
const catResponse = await fetch('/api/menu/product-categories');
categories = await catResponse.json();
window.allMenuProducts = []; // Lazy load on first search
renderCategories();
if (categories.length > 0) await loadCategoryProducts(categories[0].id);
// Preload in background
setTimeout(preloadProducts, 500);
} catch (err) {
document.getElementById('categoriesPanel').innerHTML = '<div class="error">Ошибка загрузки</div>';
}
}

}
async function loadCategoryProducts(categoryId) {
// Check cache first
if (categoryProductsCache[categoryId]) {
allProducts = categoryProductsCache[categoryId];
renderProducts();
return;
}
// Show loading
document.getElementById('productsGrid').innerHTML = '<div class="loading">Загрузка...</div>';
try {
// Load products for this category
const response = await fetch(`/api/menu/products?category_id=${categoryId}&limit=100`);
const data = await response.json();
const products = data.products || data || [];
// Process products
const processedProducts = [];
products.forEach(p => {
if (p.is_variable && p.variations && p.variations.length > 0) {
p.variations.forEach(v => {
// v.name already contains full name like "Ветчина и грибы - 30см"
if (v.price > 0) processedProducts.push({...p, variantName: v.name, variantId: v.id, price: v.price, displayName: v.name});
});
} else if (p.price > 0) {
processedProducts.push({...p, displayName: p.name});
}
});
// Cache and display
categoryProductsCache[categoryId] = processedProducts;
allProducts = processedProducts;
renderProducts();
} catch (err) {
document.getElementById('productsPanel').innerHTML = '<div class="error">Ошибка загрузки товаров</div>';
}
}

function renderProductBatch(grid, products) {
const html = products.map(p => {
const isVariant = p.variantId && p.variantId !== p.id;
const displayPrice = p.price;
const productId = p.variantId || p.id;

'has-variants' : ''}" onclick="addToCart('${p.displayName.replace(/'/g, "\\'")}', ${p.price}, ${productId}, ${parentIdArg})"><div class="product-name">${p.displayName}</div><div class="product-price">${displayPrice ? displayPrice.toFixed(2) : '0.00'} Br</div>${isVariant ? `<div class="variant-badge">вариант</div>` : ''}</div>`;
}).join('');
// Remove load more button if exists
const loadMore = grid.querySelector('.load-more-btn');
if (loadMore) loadMore.remove();
grid.insertAdjacentHTML('beforeend', html);
// Add "Load more" button if more products exist
const catProducts = allProducts || [];
const currentCount = grid.querySelectorAll('.product-card').length;
if (currentCount < catProducts.length) {
const loadMoreBtn = document.createElement('div');
loadMoreBtn.className = 'load-more-btn';
loadMoreBtn.innerHTML = `<button onclick="loadMoreProducts()" style="width:100%;padding:12px;background:#f0f0f0;border:1px solid #ddd;border-radius:6px;cursor:pointer;margin-top:10px;">Показать еще (${catProducts.length - currentCount})</button>`;
grid.appendChild(loadMoreBtn);
}
}
function loadMoreProducts() {
const grid = document.getElementById('productsGrid');
const catProducts = allProducts || [];
const currentCount = grid.querySelectorAll('.product-card').length;
const nextBatch = catProducts.slice(currentCount, currentCount + PRODUCTS_PER_PAGE);
if (nextBatch.length > 0) {
renderProductBatch(grid, nextBatch);
}
}
function onProductClick(productId) {
const products = window.allMenuProducts || [];

const hasVariants = product.is_variable && product.variations && product.variations.length > 0 && product.variations.some(v => v.price > 0);

}

// Debounce utility
function debounce(func, wait) {
let timeout;
return function(...args) {
clearTimeout(timeout);
timeout = setTimeout(() => func.apply(this, args), wait);
};
}
// Debounced search
const debouncedSearch = debounce((query) => performSearch(query), 300);
function searchProducts(query) {
debouncedSearch(query);
}
function performSearch(query) {
// Desktop search
const results = document.getElementById('searchResults');
// Mobile search
const mobileResults = document.getElementById('mobileSearchResults');
const isMobile = window.innerWidth <= 768;
const targetResults = isMobile && mobileResults ? mobileResults : results;
if (!query || query.length < 2) {
results.style.display = 'none';
if (mobileResults) mobileResults.style.display = 'none';
return;
}
// Use window.allMenuProducts for search (contains ALL products, not just current category)
const searchProducts = window.allMenuProducts || [];
const processedForSearch = [];
searchProducts.forEach(p => {
if (p.is_variable && p.variations && p.variations.length > 0) {
p.variations.forEach(v => {
// v.name already contains full name
if (v.price > 0) processedForSearch.push({...p, variantName: v.name, variantId: v.id, price: v.price, displayName: v.name});
});
} else if (p.price > 0) {
processedForSearch.push({...p, displayName: p.name});
}
});
const matches = processedForSearch.filter(p => p.displayName.toLowerCase().includes(query.toLowerCase())).slice(0, 10);
// Note: hideSearch(true, true) clears input and keeps focus for continuous search
const html = matches.length === 0 ? '<div class="search-result-item">Ничего не найдено</div>' : matches.map(p => {
const productId = p.variantId || p.id;
const parentId = p.variantId ? p.id : 'null';
return `<div class="search-result-item" onclick="addToCart('${p.displayName.replace(/'/g, "\\'")}', ${p.price}, ${productId}, ${parentId}); hideSearch(true, true)"><span>${p.displayName}</span><span style="color:#1c00bc;font-weight:600">${p.price.toFixed(2)} Br</span></div>`;
}).join('');
targetResults.innerHTML = html;
targetResults.style.display = 'block';
}
function hideSearch(clearInput = true, keepFocus = false) {

function addToCart(name, price, productId, parentId) {
const existing = cart.find(item => item.name === name && item.productId === productId);
if (existing) {
existing.qty++;
} else {
// parentId is the parent product id (for variants), null for regular products
const actualParentId = parentId === 'null' ? null : parentId;
cart.push({ name, price, qty: 1, productId, parentId: actualParentId, discount: 0 });
}
renderCart();
if (currentTab !== 'order') {
document.querySelector('.left-tab').click();
}
}
function setItemDiscount(index, discount) {
cart[index].discount = discount;
renderCart();
}
function renderCart() {
const container = document.getElementById('cartItems');
if (cart.length === 0) {
container.innerHTML = '<div class="empty-cart">Корзина пуста</div>';
} else {
let html = '';
cart.forEach((item, i) => {
const discountedPrice = item.price * (1 - item.discount / 100);
html += `<div class="cart-item">
<span class="item-name">${item.name}${item.discount > 0 ? `<span class="item-discount">-${item.discount}%</span>` : ''}</span>
<span class="item-price">${discountedPrice.toFixed(2)}</span>
<div class="item-qty-control">
<button class="qty-btn minus" onclick="changeQty(${i}, -1)">−</button>
<span class="item-qty">${item.qty}</span>
<button class="qty-btn" onclick="changeQty(${i}, 1)">+</button>
</div>
<span class="item-total">${(discountedPrice * item.qty).toFixed(2)} Br</span>
</div>`;
const itemDiscounts = window.itemDiscounts || [{name: '5%', value: 5}, {name: '10%', value: 10}];
html += `<div class="item-discount-selector">`;
if (itemDiscounts.length === 0) {
// Default if no item discounts configured
html += `<button class="item-discount-btn ${item.discount === 0 ? 'active' : ''}" onclick="setItemDiscount(${i}, 0)">0%</button>`;
html += `<button class="item-discount-btn ${item.discount === 5 ? 'active' : ''}" onclick="setItemDiscount(${i}, 5)">5%</button>`;
html += `<button class="item-discount-btn ${item.discount === 10 ? 'active' : ''}" onclick="setItemDiscount(${i}, 10)">10%</button>`;
} else {
itemDiscounts.forEach(d => {
html += `<button class="item-discount-btn ${item.discount === d.value ? 'active' : ''}" onclick="setItemDiscount(${i}, ${d.value})">${d.name}</button>`;
});
}
html += `</div>`;
});
container.innerHTML = html;
}
updateTotal();
}
function changeQty(index, delta) {
cart[index].qty += delta;
if (cart[index].qty <= 0) cart.splice(index, 1);
renderCart();
}
function updateTotal() {
const subtotal = cart.reduce((sum, item) => {
const discountedPrice = item.price * (1 - item.discount / 100);
return sum + discountedPrice * item.qty;
}, 0);
const discountType = document.getElementById('discountType').value;
const discountValue = parseFloat(document.getElementById('discountValue').value) || 0;
let discountAmount = 0;
let discountPercent = 0;
if (discountType === 'percent') {
discountPercent = discountValue;
discountAmount = subtotal * discountValue / 100;
} else {
discountAmount = discountValue;
discountPercent = subtotal > 0 ? (discountValue / subtotal * 100) : 0;
}
const total = subtotal - discountAmount;
document.getElementById('totalAmount').textContent = total.toFixed(2) + ' Br';
calculateChange(); // Update change display when total changes
}
function setReceived(amount) {
document.getElementById('receivedAmount').value = amount;
calculateChange();
}
function calculateChange() {
const received = parseFloat(document.getElementById('receivedAmount').value) || 0;
const totalText = document.getElementById('totalAmount').textContent;
const total = parseFloat(totalText.replace(' Br', '').replace(',', '.')) || 0;
const change = received - total;
const changeDisplay = document.getElementById('changeDisplay');
if (received > 0 && change >= 0) {
changeDisplay.textContent = 'Сдача: ' + change.toFixed(2) + ' Br';
changeDisplay.style.color = '#4caf50';
} else if (received > 0 && change < 0) {
changeDisplay.textContent = 'Не хватает: ' + Math.abs(change).toFixed(2) + ' Br';
changeDisplay.style.color = '#ff9800';
} else {
changeDisplay.textContent = '';
}
}
async function saveOrder() {
if (cart.length === 0) { alert('Корзина пуста!'); return; }
const subtotal = cart.reduce((sum, item) => sum + item.price * item.qty, 0);
const discountType = document.getElementById('discountType').value;
const discountValue = parseFloat(document.getElementById('discountValue').value) || 0;
let discountAmount = 0;
let discountPercent = 0;
if (discountType === 'percent') {
discountPercent = discountValue;
discountAmount = subtotal * discountValue / 100;
} else {
discountAmount = discountValue;
discountPercent = subtotal > 0 ? (discountValue / subtotal * 100) : 0;
}
const itemsTotal = cart.reduce((sum, item) => {
const discountedPrice = item.price * (1 - item.discount / 100);
return sum + discountedPrice * item.qty;
}, 0);
const total = itemsTotal - discountAmount;
// Check if editing existing order
const urlParams = new URLSearchParams(window.location.search);
const editOrderId = urlParams.get('order');
const orderData = {
customer_name: document.getElementById('clientName').value || 'Гость',
customer_phone: document.getElementById('clientPhone').value.replace(/\s/g, '') || '',
delivery_type: currentOrderType,
address: buildAddress(),
payment_type: document.getElementById('paymentSelect').value,
comment: document.getElementById('orderComment').value || '',
preorder_time: document.getElementById('datetimePicker').classList.contains('active') ? document.getElementById('preorderTime').value : null,
items: cart.map(item => {
const mapped = {
product_id: item.parentId || item.productId, // parent id for variants, same for regular
variant_id: item.productId, // actual product/variant id
quantity: item.qty,
price: item.price,
discount: item.discount
};
return mapped;
}),
subtotal_amount: subtotal,
discount_percent: discountPercent,
discount_amount: discountAmount,
delivery_fee: 0,
total_amount: total,
marks: Array.from(document.querySelectorAll('#marksContainer .mark-tag.active')).map(el => el.getAttribute('data-mark-id')).filter(Boolean)
};

console.log('Selected marks:', orderData.marks);
console.log('Active mark elements:', document.querySelectorAll('#marksContainer .mark-tag.active'));

try {
const url = editOrderId ? `/api/orders/${editOrderId}` : '/api/orders/';
const method = editOrderId ? 'PUT' : 'POST';
const response = await fetch(url, {
method: method,
headers: { 'Content-Type': 'application/json' },
body: JSON.stringify(orderData)
});
if (response.ok) {
// Очищаем форму для нового заказа
window.location.href = 'crm-operator-workplace.html';
} else {
const error = await response.json();
alert('Ошибка: ' + (error.detail || 'Не удалось сохранить заказ'));
}
} catch (err) {
alert('Ошибка сохранения: ' + err.message);
}
}
